La state


class Button extends React.Component {
state = {
clicked: false
}

constructor() {
super();
this.handleClick = this.handleClick.bind(this);
}

handleClick() {
this.setState({ clicked: !this.state.clicked });
}

render() {
return (
  <div
    className="myButton"
    style={{ backgroundColor: this.state.clicked ? 'red' : 'blue' }}
    onClick={this.handleClick}
  >
    {this.props.text}
  </div>
);
}
}

ReactDOM.render(<Button text="Changement" />, document.getElementById('buttonReactState'));
